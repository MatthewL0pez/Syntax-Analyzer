function square(num integer)
{
    return num * num;
}
function choose(flag boolean, left, right integer)
{
    if (flag == true)
        return left;
    otherwise
        return right;
    fi
}
@
integer start, end, step, current, chosen;
boolean useStart;
real sample;
@
read(start, end, step);
current = start;
useStart = true;
sample = 4.5;
while (current <= end)
{
    write(current);
    chosen = choose(useStart, current, end);
    write(chosen);
    write(square(current));
    current = current + step;
}
